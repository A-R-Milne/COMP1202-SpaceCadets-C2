#A formula to calculate a power, and then show in standard form: X^Y = Z*10^C
clear X;
assign X, 5;
clear Y;
assign Y, 3;
clear Z;
assign Z, X;
decr Y;
while Y not 0 do;
	decr Y;
	* Z, X, Z;
end;
clear A;
assign A, 10;
clear B;
assign B, 1;
clear C;
clear D;
assign D, 1;
while D not 0 do;
	if Z > A;
		/ Z, A, Z;
		incr C;
	else;
		if Z < B;
			* Z, A, Z;
			decr C;
		else;
			decr D;
		end;
	end;
end;
